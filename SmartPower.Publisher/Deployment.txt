# Install dependencies
sudo apt update && sudo apt install -y libicu-dev
sudo apt update && sudo apt install -y build-essential libnserial-dev

# For testing purposes
sudo apt update
sudo apt install -y python3-serial
nano test_usb.py


# Publish
dotnet publish -c Release -r linux-arm64 --self-contained true -p:PublishSingleFile=true

# Stop service
sudo systemctl stop smartpower

# Clean old files
cd /opt/smartpower
ls
rm -rf *
ls

# Copy
scp -O -r * root@192.168.1.182:/opt/smartpower

# Install
sudo chmod +x /opt/smartpower/SmartPower.Publisher

sudo nano /etc/systemd/system/smartpower.service

# Service file contents
[Unit]
Description=SmartPower Worker Service
After=network.target

[Service]
Type=notify
ExecStart=/opt/smartpower/SmartPower.Publisher
WorkingDirectory=/opt/smartpower
User=root
Restart=always
RestartSec=10

[Install]
WantedBy=multi-user.target

# Load the new file
sudo systemctl daemon-reload

# Enable it to start on boot
sudo systemctl enable smartpower

# Start it right now
sudo systemctl start smartpower

# Check status
systemctl status smartpower

# To manual debug
cd /opt/smartpower
./SmartPower.Publisher

# Resolve DietPi.local
apt update
apt install avahi-daemon

systemctl enable avahi-daemon
systemctl start avahi-daemon

# Diagnostics
screen /dev/ttyUSB0 9600
